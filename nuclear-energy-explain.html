def start_chain_reaction(uranium_atoms):
    energy_output = 0
    neutrons = 1 # Start with 1 trigger neutron

    while uranium_atoms > 0:
        # One neutron splits one atom
        split_atoms = neutrons
        uranium_atoms -= split_atoms
        
        # E=mc^2 energy release
        energy_output += split_atoms * (C ** 2)
        
        # Each split creates 3 new neutrons
        neutrons = split_atoms * 3
        
        if energy_output > SAFE_LIMIT:
            control_rods.activate() # Stop explosion!
            break
            
    return energy_output
